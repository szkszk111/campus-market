<script>

const supabase = Supabase.createClient(SUPABASE_URL, SUPABASE_ANON_KEY);

const DB = {
    async registerUser(user) {
        const { error } = await supabase.from("users").insert([user]);
        return !error;
    },
    async getUserByUsername(username) {
        const { data } = await supabase.from("users").select("*").eq("username", username).maybeSingle();
        return data;
    },
    async updateUser(id, data) {
        const { error } = await supabase.from("users").update(data).eq("id", id);
        return !error;
    },
    async getProducts() {
        const { data } = await supabase.from("products").select("*").order("id", { ascending: false });
        return data || [];
    },
    async addProduct(product) {
        const { error } = await supabase.from("products").insert([product]);
        return !error;
    },
    async getProductById(id) {
        const { data } = await supabase.from("products").select("*").eq("id", id).maybeSingle();
        return data;
    },
    async updateProductStatus(id, status) {
        const { error } = await supabase.from("products").update({ status }).eq("id", id);
        return !error;
    },
    async addOrder(order) {
        const { error } = await supabase.from("orders").insert([order]);
        return !error;
    },
    async getOrders() {
        const { data } = await supabase.from("orders").select("*").order("id", { ascending: false });
        return data || [];
    }
};

const App = {
    currentUser: null,
    async init() {
        const user = localStorage.getItem("currentUser");
        if (user) {
            try { this.currentUser = JSON.parse(user); } catch (e) {}
        }
        this.updateNav();
        this.loadProducts();
        this.bindEvents();
    },
    updateNav() {
        const loginBtn = document.getElementById("loginBtn");
        const logoutBtn = document.getElementById("logoutBtn");
        const userBtn = document.getElementById("userBtn");
        if (this.currentUser) {
            loginBtn.style.display = "none";
            logoutBtn.style.display = "inline-block";
            userBtn.style.display = "inline-block";
        } else {
            loginBtn.style.display = "inline-block";
            logoutBtn.style.display = "none";
            userBtn.style.display = "none";
        }
    },
    navigate(pageId) {
        document.querySelectorAll(".page").forEach(p => p.classList.remove("active"));
        document.getElementById(`page-${pageId}`).classList.add("active");
        if (pageId === "user") this.loadUserPage();
    },
    showModal(modalId) { document.getElementById(modalId).classList.add("active"); },
    closeModal(modalId) { document.getElementById(modalId).classList.remove("active"); },
    switchModal(from, to) { this.closeModal(from); this.showModal(to); },
    async loadProducts() {
        const keyword = document.getElementById("searchInput").value.toLowerCase();
        const category = document.getElementById("categorySelect").value;
        const list = await DB.getProducts();
        let filter = list;
        if (keyword) filter = filter.filter(p => p.title.toLowerCase().includes(keyword));
        if (category) filter = filter.filter(p => p.category === category);
        const container = document.getElementById("productList");
        if (filter.length === 0) {
            container.innerHTML = `<div class="empty">暂无商品，快来发布吧！</div>`;
            return;
        }
        container.innerHTML = filter.map(p => `
            <div class="product-card" onclick="App.showDetail(${p.id})">
                <div class="product-img"></div>
                <div class="product-info">
                    <div class="product-title">${p.title}</div>
                    <div class="product-price">¥${p.price}</div>
                    <div class="product-category">${p.category} | ${p.status === 'onsale' ? '在售' : '已售'}</div>
                </div>
            </div>
        `).join("");
    },
    async showDetail(id) {
        const p = await DB.getProductById(id);
        if (!p) { this.showToast("商品不存在", "error"); return; }
        const seller = await DB.getUserByUsername(p.seller_id);
        const container = document.getElementById("detailContent");
        let btn = "";
        if (this.currentUser && this.currentUser.username === p.seller_id) {
            btn = `<button class="btn btn-default" disabled>我发布的商品</button>`;
        } else if (p.status === "onsale") {
            btn = `<button class="btn btn-success" onclick="App.buyProduct(${p.id})">立即购买</button>`;
        } else {
            btn = `<button class="btn btn-default" disabled>已售出</button>`;
        }
        container.innerHTML = `
            <h2>${p.title}</h2>
            <div class="detail-price">¥${p.price}</div>
            <div class="detail-info">
                <p>分类：${p.category}</p>
                <p>原价：¥${p.original_price || '无'}</p>
                <p>卖家：${seller?.nickname || p.seller_id}</p>
                <p>电话：${seller?.phone || '未填写'}</p>
                <p>校区：${seller?.campus || '未填写'}</p>
                <p>描述：${p.description || '无描述'}</p>
            </div>
            ${btn}
        `;
        this.navigate("detail");
    },
    goPublish() {
        if (!this.currentUser) {
            this.showToast("请先登录", "error");
            this.showModal("loginModal");
            return;
        }
        this.navigate("publish");
    },
    bindEvents() {
        document.getElementById("loginForm").addEventListener("submit", async (e) => {
            e.preventDefault();
            const form = new FormData(e.target);
            const data = Object.fromEntries(form);
            const user = await DB.getUserByUsername(data.username);
            if (!user || user.password !== data.password) {
                this.showToast("账号或密码错误", "error");
                return;
            }
            this.currentUser = user;
            localStorage.setItem("currentUser", JSON.stringify(user));
            this.updateNav();
            this.closeModal("loginModal");
            this.showToast("登录成功", "success");
        });

        document.getElementById("registerForm").addEventListener("submit", async (e) => {
            e.preventDefault();
            const form = new FormData(e.target);
            const data = Object.fromEntries(form);
            const exists = await DB.getUserByUsername(data.username);
            if (exists) {
                this.showToast("用户名已存在", "error");
                return;
            }
            const ok = await DB.registerUser(data);
            if (ok) {
                this.showToast("注册成功，请登录", "success");
                this.switchModal("registerModal", "loginModal");
            } else {
                this.showToast("注册失败", "error");
            }
        });

        document.getElementById("publishForm").addEventListener("submit", async (e) => {
            e.preventDefault();
            if (!this.currentUser) { this.showToast("请登录", "error"); return; }
            const form = new FormData(e.target);
            const data = Object.fromEntries(form);
            data.seller_id = this.currentUser.username;
            data.status = "onsale";
            const ok = await DB.addProduct(data);
            if (ok) {
                this.showToast("发布成功", "success");
                e.target.reset();
                this.navigate("home");
                this.loadProducts();
            } else {
                this.showToast("发布失败", "error");
            }
        });
    },
    async buyProduct(id) {
        if (!this.currentUser) { this.showToast("请登录", "error"); return; }
        const p = await DB.getProductById(id);
        if (!p || p.status !== "onsale") { this.showToast("商品已售出", "error"); return; }
        if (p.seller_id === this.currentUser.username) { this.showToast("不能购买自己的商品", "error"); return; }
        await DB.updateProductStatus(id, "sold");
        await DB.addOrder({
            product_id: id,
            buyer_id: this.currentUser.username,
            seller_id: p.seller_id,
            status: "done"
        });
        this.showToast("购买成功！", "success");
        this.showDetail(id);
    },
    async loadUserPage() {
        if (!this.currentUser) return;
        document.getElementById("userUsername").value = this.currentUser.username;
        document.getElementById("userNickname").value = this.currentUser.nickname || "";
        document.getElementById("userPhone").value = this.currentUser.phone || "";
        document.getElementById("userCampus").value = this.currentUser.campus || "";
        this.loadUserProducts();
        this.loadOrders();
    },
    showUserTab(tab) {
        document.querySelectorAll(".tab-btn").forEach(b => b.classList.remove("active"));
        document.querySelectorAll(".tab-content").forEach(c => c.classList.remove("active"));
        event.target.classList.add("active");
        document.getElementById(`tab-${tab}`).classList.add("active");
    },
    async updateUserInfo() {
        if (!this.currentUser) return;
        const data = {
            nickname: document.getElementById("userNickname").value,
            phone: document.getElementById("userPhone").value,
            campus: document.getElementById("userCampus").value
        };
        const ok = await DB.updateUser(this.currentUser.id, data);
        if (ok) {
            this.currentUser = { ...this.currentUser, ...data };
            localStorage.setItem("currentUser", JSON.stringify(this.currentUser));
            this.showToast("保存成功", "success");
        } else {
            this.showToast("保存失败", "error");
        }
    },
    async loadUserProducts() {
        if (!this.currentUser) return;
        const list = await DB.getProducts();
        const mine = list.filter(p => p.seller_id === this.currentUser.username);
        const container = document.getElementById("userProductList");
        if (mine.length === 0) {
            container.innerHTML = `<div class="empty">暂无发布商品</div>`;
            return;
        }
        container.innerHTML = mine.map(p => `
            <div style="padding:10px; border-bottom:1px solid #eee;">
                <p>${p.title} - ¥${p.price} (${p.status === 'onsale' ? '在售' : '已售'})</p>
            </div>
        `).join("");
    },
    async loadOrders() {
        if (!this.currentUser) return;
        const list = await DB.getOrders();
        const myOrders = list.filter(o =>
            o.buyer_id === this.currentUser.username ||
            o.seller_id === this.currentUser.username
        );
        const container = document.getElementById("orderList");
        if (myOrders.length === 0) {
            container.innerHTML = `<div class="empty">暂无订单</div>`;
            return;
        }
        container.innerHTML = myOrders.map(o => `
            <div style="padding:10px; border-bottom:1px solid #eee;">
                <p>商品ID：${o.product_id} | ${o.buyer_id === this.currentUser.username ? '我购买的' : '我卖出的'}</p>
            </div>
        `).join("");
    },
    logout() {
        this.currentUser = null;
        localStorage.removeItem("currentUser");
        this.updateNav();
        this.navigate("home");
        this.showToast("退出成功", "success");
    },
    showToast(msg, type = "default") {
        const toast = document.getElementById("toast");
        toast.textContent = msg;
        toast.className = "toast show " + type;
        setTimeout(() => toast.className = "toast", 2000);
    }
};

App.init();
